# 大图预览（点击放大弹出，关闭缩回原位置）

调用方式

	<PopCover ref="pop"></PopCover>

			this.$refs.pop.setTransformOrigin(perx*100+'%'+' '+pery*100+'%','图片地址');

PopCover.vue

<template>
	<div class="start" v-bind:class="{'ani':ani!=''}" v-bind:style="{ 
        'transformOrigin':transformOrigin,
        'transform':transform
    }">
  
    	<iframe ref="iframe" src="/pop" width="100%" height="100%" frameborder="0"></iframe>
    </div>
</template>
<script>
	export default{
		components:{},
		props:[],
		data(){
			return {
				t:'start',
	            left:'0px',
	            top:'0px',
	            width:'10px',
	            height:'10px',
	            ani:'',

	            transformOrigin:'50% 50%',
	            transform:'translate3d(0,0,0) scale(0)',


	           	doing:false,  //是否正在动画过程中

	           	status:'mini'  //none：没显示 normal:已显示
			}
		},
		methods:{
			setTransformOrigin:function(v,imgUrl){

				if(this.doing && (this.status == 'normal')){
					return false;
				}

				document.body.style.overflow = 'hidden';

				this.ani = '';
				this.doing = true;
				this.status = 'normal'

				this.transformOrigin = v;

				//给iframe发消息
				this.$refs.iframe.contentWindow.postMessage(imgUrl,'*');

				setTimeout(_=>{

					this.setTransform();

				},200)
			},
			setTransform:function(){  
 
				this.ani = 'ani';
				this.transform = 'translate3d(0,0,0) scale(1)';

				setTimeout(_=>{

					this.doing = false;

				},1000)
			},
			fadeOut:function(){

				if(this.doing){
					return false;
				}

				this.doing = true;
				this.transform = 'translate3d(0,0,0) scale(0)'


				setTimeout(_=>{
					document.body.style.overflow = 'auto';

					this.ani = '';
					this.doing = false;
					this.status = 'none';
					this.$refs.iframe.contentWindow.postMessage('clear','*');


				},1100);

			}
		},
		mounted(){

			window.addEventListener("message", (e)=>
			{
				if(e.data == 'close'){
					this.fadeOut();
				}
			})
		},
		created(){

		},
		computed:{

		},
		watch:{

		}
	}

</script>
<style scoped>

    position: fixed;
    left:0;
    top:0;

    width: 100%;
    height: 100%;

}

 transition: all 0.8s;

}

	border:none;
	overflow: hidden;
	-webkit-transform: translate3d(0,0,0);
	-ms-transform: translate3d(0,0,0);
	-o-transform: translate3d(0,0,0);
	transform: translate3d(0,0,0);
}
</style>

Pop.vue

<template>
	<div>

		 <el-image-viewer
        v-if="showViewer"
        :on-close="close"
        :url-list="viewUrl" 
         />
	</div>
</template>
<script>
import ElImageViewer from 'element-ui/packages/image/src/image-viewer';

	export default{
		components:{ElImageViewer},
		props:[],
		data(){
			return {
				showViewer:false,
				viewUrl:[]
			}
		},
		methods:{
			close:function(){
				window.parent.postMessage('close','*');
			}
		},
		mounted(){

			window.addEventListener("message", (e)=>
			{
				if(e.data == 'clear'){
					this.showViewer = false;
					this.viewUrl = [];
				}else{

				this.viewUrl = [e.data];
				this.showViewer = true;
				}
			})


		},
		created(){

		},
		computed:{

		},
		watch:{

		}
	}

</script>
