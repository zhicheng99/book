# input失去焦点自动格式为两位浮点数Vue指令


Vue.directive("floatTofixed", {  //格式成两位小数
    inserted: function(el, binding, vnode) {

        var input = el.children[1].children[0].children[0];
        input.addEventListener("blur", function(e) {
            if(input.value!=''){
              input.value  = (input.value*1).toFixed(2);
              // vnode.data.model.callback(input.value);

              input.dispatchEvent(new Event('input'))
            }
        })

    }
})
